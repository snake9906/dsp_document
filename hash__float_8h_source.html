26 //
27 // Only used for floats with known iec559 floats, and certain values in
28 // numeric_limits

30 inline auto hash_binary(char *ptr, std::size_t length) -> std::size_t {
31 std::size_t seed = 0;
32
33 if (length >= sizeof(std::size_t)) {
34 std::memcpy(&seed, ptr, sizeof(std::size_t));
35 length -= sizeof(std::size_t);
36 ptr += sizeof(std::size_t);
37
38 while (length >= sizeof(std::size_t)) {
39 std::size_t buffer = 0;
40 std::memcpy(&buffer, ptr, sizeof(std::size_t));
41 hash_float_combine(seed, buffer);
42 length -= sizeof(std::size_t);
43 ptr += sizeof(std::size_t);
44 }
45 }
46
47 if (length > 0) {
48 std::size_t buffer = 0;
49 std::memcpy(&buffer, ptr, length);
50 hash_float_combine(seed, buffer);
51 }
52
53 return seed;
54 }

95 inline auto float_hash_value(T v) -> std::size_t {
96 switch (std::fpclassify(v)) {
97 case FP_ZERO:
98 return 0;
99 case FP_INFINITE:
100 return (std::size_t)(v > 0 ? -1 : -2);
101 case FP_NAN:
102 return (std::size_t)(-3);
103 case FP_NORMAL:
104 case FP_SUBNORMAL:
105 return float_hash_impl(v, 0);
106 default:
107 // BOOST_ASSERT(0);
108 return 0;
109 }
