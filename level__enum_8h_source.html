13namespace dsp {
14namespace serialization {
15 enum level_type {
16 // Don't serialize this type. An attempt to do so should
17 // invoke a compile time assertion.
18 not_serializable = 0,
19 // write/read this type directly to the archive. In this case
20 // serialization code won't be called. This is the default
21 // case for fundamental types. It presumes a member function or
22 // template in the archive class that can handle this type.
23 // there is no runtime overhead associated reading/writing
24 // instances of this level
25 primitive_type = 1,
26 // Serialize the objects of this type using the objects "serialize"
27 // function or template. This permits values to be written/read
28 // to/from archives but includes no class or version information.
29 object_serializable = 2,
31 // once an object is serialized at one of the above levels, the
32 // corresponding archives cannot be read if the implementation level
33 // for the archive object is changed.
35 // Add class information to the archive. Class information includes
36 // implementation level, class version and class name if available
37 object_class_info = 3
38 };
