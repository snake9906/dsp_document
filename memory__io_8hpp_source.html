317 auto pop_impl(void *des, size_t sz) -> bool {
318 if (m_memory_blocks.empty()) { return false; }
319
320 auto block = m_memory_blocks.front();
321 if (block.size != sz) { return false; }
322
323 m_memory_blocks.pop();
324 memcpy(des, &m_memory[0] + block.pos, sz);
325 m_end -= sz;
326
327 return true;
328 }

330 void push_impl(const void *src, size_t sz) {
331 if (m_memory.size() < sz + m_end) { m_memory.resize(cal_size(sz + m_end)); }
332
333 memcpy(&m_memory[0] + m_end, src, sz);
334 m_memory_blocks.emplace(m_end, sz);
335
336 m_end += sz;
337 }
