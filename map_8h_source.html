34namespace detail {
35
36template<class M, class T>
37struct mp_map_replace_impl;
38
39template<template<class...> class M, class... U, class T>
40struct mp_map_replace_impl<M<U...>, T> {
41 using K = mp_first<T>;
42
43 // mp_replace_if is inlined here using a struct _f because of msvc-14.0
44
45 template<class V>
46 struct _f {
47 using type = mp_if<std::is_same<mp_first<V>, K>, T, V>;
48 };
49
50 using type = mp_if<mp_map_contains<M<U...>, K>, M<typename _f<U>::type...>, M<U..., T>>;
51};
52
53}// namespace detail

58// mp_map_update<M, T, F>
59namespace detail {
60
61template<class M, class T, template<class...> class F>
62struct mp_map_update_impl {
63 template<class U>
64 using _f = std::is_same<mp_first<T>, mp_first<U>>;
65
66 // _f3<L<X, Y...>> -> L<X, F<X, Y...>>
67 template<class L>
68 using _f3 = mp_assign<L, mp_list<mp_first<L>, mp_rename<L, F>>>;
69
70 using type = mp_if<mp_map_contains<M, mp_first<T>>, mp_transform_if<_f, _f3, M>, mp_push_back<M, T>>;
71};
72
73}// namespace detail
