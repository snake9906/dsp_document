184 template<typename ARRAY_T>
185 void init_array_data() {
186 auto attr = m_p_conn->get_attr_by_oid<ARRAY_T>(m_oid, m_field_id);
187 if (attr.coll_num == 0) { return; }
188
189 if (m_mode == CLIENT_OPEN) {
190 m_data = dsp::make_span(m_p_conn->get_pt_object_by_oid<T *>(attr.first_element_oid), attr.coll_num);
191 return;
192 }
193
194 if (m_mode == CS_OPEN) {
195 m_cs_data = m_p_conn->get_related_objects<T>(m_oid, m_field_id);
196 m_data = dsp::make_span(m_cs_data);
197 return;
198 }
199 }

205 void init_list_data() {
206 auto attr = m_p_conn->get_attr_by_oid<s_Dlist>(m_oid, m_field_id);
207 if (attr.number == 0) { return; }
208
209 if (m_mode == CLIENT_OPEN) {
210 m_list_data.resize(attr.number);
211 T *p_list_data{nullptr};
212
213 for (int i = 0; i < m_list_data.size(); ++i) {
214 if (i == 0) {
215 p_list_data = m_p_conn->get_pt_object_by_oid<T>(attr.first_element);
216 } else {
217 p_list_data = m_p_conn->get_pt_object_by_oid<T>(
218 m_p_conn->get_attr_by_oid<s_Dlist_mate>(p_list_data->oid(), m_child_field_id).next_element);
219 }
220
221 m_list_data[i] = p_list_data;
222 }
223 m_data = dsp::make_span(m_list_data);
224
225 return;
226 }
227
228 if (m_mode == CS_OPEN) {
229 m_cs_data.resize(attr.number);
230
231 for (int i = 0; i < m_cs_data.size(); ++i) {
232 if (i == 0) {
233 m_cs_data[i] = m_p_conn->get_object_by_oid<T>(attr.first_element);
234 } else {
235 m_cs_data[i] = m_p_conn->get_object_by_oid<T>(
236 m_p_conn->get_attr_by_oid<s_Dlist_mate>(m_cs_data[i - 1].oid(), m_child_field_id)
237 .next_element);
238 }
239 }
240 m_data = dsp::make_span(m_cs_data);
241
242 return;
243 }
244 }

249 void init_child_info() {
250 auto attr = m_p_conn->get_attr_by_oid<s_Dlist>(m_oid, m_field_id);
251 if (attr.number == 0) { return; }
252
253 m_child_class_id = attr.first_element.db_class_id;
254 auto relation_infos = m_p_conn->get_all_relations_info(m_child_class_id);
255 std::any_of(relation_infos.begin(), relation_infos.end(), [&](db_relation_info &info) -> bool {
256 if (info.field_id_out == m_field_id) {
257 m_child_class_id = info.field_id_in;
258 return true;
259 }
260 return false;
261 });
262 }
