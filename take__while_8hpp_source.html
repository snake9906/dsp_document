42template<class Predicate, class Iterator>
43class while_iterator : public detail::while_iterator_base<Predicate, Iterator>::type {
44 using super_t = typename detail::while_iterator_base<Predicate, Iterator>::type;
45
46 friend class iterator_core_access;
47
48public:
49 while_iterator() = default;
50
51 while_iterator(Predicate f, Iterator x, Iterator end) : super_t(x), m_predicate(f), m_x(x), m_end(end) {
52 if (not m_predicate(*m_x)) { m_x = m_end; }
53 }
54
55 explicit while_iterator(Iterator x) : super_t(x), m_predicate() {
56 // Pro8 is a little too aggressive about instantiating the
57 // body of this function.
58#if !BOOST_WORKAROUND(__MWERKS__, BOOST_TESTED_AT(0x3003))
59 // Don't allow use of this constructor if Predicate is a
60 // function pointer type, since it will be 0.
61 BOOST_STATIC_ASSERT(is_class<Predicate>::value);
62#endif
63 }
64
65 template<class OtherIterator>
66 explicit while_iterator(while_iterator<Predicate, OtherIterator> const &t,
67 typename enable_if_convertible<OtherIterator, Iterator>::type * /*unused*/ = 0)
68 : super_t(t.base()), m_predicate(t.predicate()), m_x(t.m_x), m_end(t.m_end) {}
69
70 auto predicate() const -> Predicate { return m_predicate; }
71
72private:
73 auto dereference() const -> typename super_t::reference { return *m_x; }
74
75 void increment() {
76 ++m_x;
77 if (not m_predicate(*m_x)) { m_x = m_end; }
78 }
79
80 friend auto operator==(const while_iterator &a, const while_iterator &b) -> bool {
81 return !static_cast<bool>(a.m_x != b.m_x or a.m_end != b.m_end);
82 }
83
84 friend auto operator!=(const while_iterator &a, const while_iterator &b) -> bool { return !(a == b); }
85
86 Predicate m_predicate;
87 Iterator m_x;
88 Iterator m_end;
89};

116template<class P, class R>
117struct while_range
118 : boost::iterator_range<boost::while_iterator<typename default_constructible_unary_fn_gen<P, bool>::type,
119 typename range_iterator<R>::type>> {
120private:
121 using base = boost::iterator_range<boost::while_iterator<typename default_constructible_unary_fn_gen<P, bool>::type,
122 typename range_iterator<R>::type>>;
123
124public:
125 using pred_t = typename default_constructible_unary_fn_gen<P, bool>::type;
126
127 while_range(P p, R &r)
128 : base(make_while_iterator(pred_t(p), boost::begin(r), boost::end(r)),
129 make_while_iterator(pred_t(p), boost::end(r))) {}
130};
