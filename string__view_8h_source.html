446 auto find(basic_string_view s, size_type pos = 0) const noexcept -> size_type {
447 if (pos > size()) { return npos; }
448
449 if (s.empty()) { return pos; }
450
451 if (s.size() > size() - pos) { return npos; }
452
453 const charT *cur = ptr_ + pos;
454 const charT *last = cend() - s.size() + 1;
455 for (; cur != last; ++cur) {
456 cur = traits::find(cur, last - cur, s[0]);
457 if (!cur) { return npos; }
458 if (traits::compare(cur, s.cbegin(), s.size()) == 0) { return cur - ptr_; }
459 }
460 return npos;
461 }

510 auto _split(basic_string_view k, bool keep_empty_parts = false) -> std::vector<basic_string_view> {
511 std::vector<basic_string_view> res{};
512
513 size_type begin_pos = 0;
514 size_type end_pos = 0;
515 while (true) {
516 end_pos = find(k, begin_pos);
517 if (end_pos == npos) {
518 res.emplace_back(substr(begin_pos));
519 break;
520 }
521
522 DEFER { begin_pos = end_pos + k.size(); };
523
524 if (begin_pos == end_pos && !keep_empty_parts) { continue; }
525 res.emplace_back(substr(begin_pos, end_pos).data(), (end_pos - begin_pos));
526 }
527
528 return res;
529 }

554 static auto join(const std::vector<basic_string_view> &string_views,
555 const std::basic_string<charT, traits> &sep = "") -> std::basic_string<charT, traits> {
556 size_t sz{0};
557 for (auto &sv : string_views) { sz += sv.size(); }
558 sz += (string_views.size() - 1) * sep.size();
559
560 std::basic_string<charT, traits> str(sz, '0');
561
562 charT *data = &str[0];
563 for (size_t i = 0; i < string_views.size(); ++i) {
564 memcpy(data, string_views[i].data(), string_views[i].size());
565 data += string_views[i].size();
566
567 if (i != string_views.size() - 1 && !sep.empty()) {
568 memcpy(data, sep.data(), sep.length());
569 data += sep.length();
570 }
571 }
572
573 return str;
574 }

802 auto rfind(basic_string_view s, size_type pos = npos) const noexcept -> size_type {
803 if (len_ < s.len_) { return npos; }
804 if (pos > len_ - s.len_) pos = len_ - s.len_;
805 if (s.len_ == 0u) {

807 return pos;

809 for (const charT *cur = ptr_ + pos;; --cur) {
810 if (traits::compare(cur, s.ptr_, s.len_) == 0) { return cur - ptr_; }
811 if (cur == ptr_) { return npos; }
812 };
