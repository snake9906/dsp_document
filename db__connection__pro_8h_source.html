212 template<typename T>
213 auto get_all_objects(uint16_t tid, size_t max_sz) -> vector<T> {
214
215 auto count = p_conn->get_lv_of_class(tid);
216 if (count <= 0) { return vector<T>{}; }
217
218 auto res = dsp::make_empty_objs<T>(count);
219
220 auto data_sz = sizeof(T) * count;
221 if (data_sz <= max_sz) {
222 p_conn->get_all_objects(tid, count, res.data(), data_sz);
223 return res;
224 }
225
226 size_t once_count = std::floor(max_sz / sizeof(T));
227 size_t x = count / once_count;
228 size_t y = count % once_count;
229
230 for (size_t i = 0; i < x; i++) {
231 p_conn->get_objects_by_sub(tid,
232 once_count * i,
233 once_count,
234 res.data() + once_count * i,
235 sizeof(T) * once_count);
236 }
237 p_conn->get_objects_by_sub(tid, once_count * x, y, res.data() + once_count * x, sizeof(T) * y);
238
239 return res;
240 }

317 template<typename T>
318 auto get_related_objects(const OID &oid, uint16_t fid, size_t max_sz = DEFAULT_MAX_MSG_SIZE) -> vector<T> {
319
320 auto info = get_attrs_info_by_id(oid.db_class_id, fid);
321 size_t count{0};
322 switch (info.type_id) {
323 case DT_S_REF_SIM:
324 case DT_S_REF_REL:
325 case DT_S_REF_INT:
326 case DT_S_REF_MATE: {
327 count = 1;
328 break;
329 }
330 case DT_S_DARRAY_COMP:
331 case DT_S_ARRAY_COMP: {
332 auto comp = get_attr_by_oid<s_array_composition>(oid, fid);
333 count = comp.coll_num;
334 break;
335 }
336 case DT_S_MARRAY_COMP: {
337 auto comp = get_attr_by_oid<s_Marray_composition>(oid, fid);
338 count = comp.coll_num;
339 break;
340 }
341 default: {
342 break;
343 }
344 }
345
346 if (count == 0) { return vector<T>{}; }
347
348 auto res = dsp::make_empty_objs<T>(count);
349 size_t data_sz = sizeof(T) * count;
350 if (data_sz <= max_sz) {
351 p_conn->get_related_objects(oid, fid, res.data(), data_sz, 0, count);
352 return res;
353 }
354
355 size_t once_count = std::floor(max_sz / sizeof(T));
356 size_t x = count / once_count;
357 size_t y = count % once_count;
358 for (int i = 0; i < x; ++i) {
359 p_conn->get_related_objects(oid,
360 fid,
361 res.data() + i * once_count,
362 sizeof(T) * once_count,
363 i * once_count,
364 once_count);
365 }
366 p_conn->get_related_objects(oid, fid, res.data() + x * once_count, sizeof(T) * y, x * once_count, y);
367
368 return res;
369 }
