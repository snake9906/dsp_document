452 auto find(basic_string_view s, size_type pos = 0) const noexcept -> size_type {
453 if (pos > size()) { return npos; }
454
455 if (s.empty()) { return pos; }
456
457 if (s.size() > size() - pos) { return npos; }
458
459 const charT *cur = ptr_ + pos;
460 const charT *last = cend() - s.size() + 1;
461 for (; cur != last; ++cur) {
462 cur = traits::find(cur, last - cur, s[0]);
463 if (!cur) { return npos; }
464 if (traits::compare(cur, s.cbegin(), s.size()) == 0) { return cur - ptr_; }
465 }
466 return npos;
467 }

516 auto _split(basic_string_view k, bool keep_empty_parts = false) -> std::vector<basic_string_view> {
517 std::vector<basic_string_view> res{};
518
519 size_type begin_pos = 0;
520 size_type end_pos = 0;
521 while (true) {
522 end_pos = find(k, begin_pos);
523 if (end_pos == npos) {
524 res.emplace_back(substr(begin_pos));
525 break;
526 }
527
528 DSP_DEFER { begin_pos = end_pos + k.size(); };
529
530 if (begin_pos == end_pos && !keep_empty_parts) { continue; }
531 res.emplace_back(substr(begin_pos, end_pos).data(), (end_pos - begin_pos));
532 }
533
534 return res;
535 }

560 static auto join(const std::vector<basic_string_view> &string_views,
561 const std::basic_string<charT, traits> &sep = "") -> std::basic_string<charT, traits> {
562 size_t sz{0};
563 for (auto &sv : string_views) { sz += sv.size(); }
564 sz += (string_views.size() - 1) * sep.size();
565
566 std::basic_string<charT, traits> str(sz, '0');
567
568 charT *data = &str[0];
569 for (size_t i = 0; i < string_views.size(); ++i) {
570 memcpy(data, string_views[i].data(), string_views[i].size());
571 data += string_views[i].size();
572
573 if (i != string_views.size() - 1 && !sep.empty()) {
574 memcpy(data, sep.data(), sep.length());
575 data += sep.length();
576 }
577 }
578
579 return str;
580 }

593 static auto join(const std::vector<std::basic_string<charT, traits>> &strings,
594 const std::basic_string<charT, traits> &sep = "") -> std::basic_string<charT, traits> {
595 std::vector<basic_string_view> views(strings.size());
596 std::transform(strings.begin(),
597 strings.end(),
598 views.begin(),
599 [](const std::basic_string<charT, traits> &str) -> basic_string_view {
600 return basic_string_view(str);
601 });
602
603 return join(views, sep);
604 }

806 auto rfind(basic_string_view s, size_type pos = npos) const noexcept -> size_type {
807 if (len_ < s.len_) { return npos; }
808 if (pos > len_ - s.len_) pos = len_ - s.len_;
809 if (s.len_ == 0u) {
810 // an empty string is always found
811 return pos;
812 }
813 for (const charT *cur = ptr_ + pos;; --cur) {
814 if (traits::compare(cur, s.ptr_, s.len_) == 0) { return cur - ptr_; }
815 if (cur == ptr_) { return npos; }
816 };
817 }
