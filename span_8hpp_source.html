156class span {
157public:
158 using element_type = T;

160 using size_type = std::size_t;
161 using difference_type = std::ptrdiff_t;
162 using pointer = T *;
163 using const_pointer = const T *;
164 using reference = T &;
165 using const_reference = const T &;
166 using iterator = T *;
167 // using const_iterator = const T *;

169 using const_reverse_iterator = std::reverse_iterator<const T *>;
