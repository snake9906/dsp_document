184template<typename T, size_t N = sizeof(T)>
185void swap_endian_impl(T &val, sizeOk) {
186 union bytesN {
187 T val;
188 uint8_t arr[N];
189 };
190
191 bytesN tmp;
192 tmp.val = val;
193 size_t n = 0;
194 while (n < N / 2) {
195 std::swap(tmp.arr[n], tmp.arr[N - n - 1]);
196 n += 1;
197 }
198
199 val = tmp.val;
200}

235template<typename F, typename... Args>
236auto func_avg_elapsed(size_t n, F &&f, Args &&...args) -> double {
237 if (n == 0) { return 0.; }
238
239 using namespace dsp::range;
240
241 // clang-format off
242 auto duration =
243 view::irange(n) |
244 action::reduce([&](double &duration, int) { duration += func_elapsed(std::forward<F>(f), std::forward<Args>(args)...); },
245 0.);
246 // clang-format on
247
248 return duration / n;
249}

257static void sleep(size_t ms, bool high_precise = false) {
258 if (ms >= 1000) { high_precise = false; }
259
260 if (!high_precise) {
261 std::this_thread::sleep_for(std::chrono::milliseconds(ms));
262 return;
263 }
264
265 static double estimate = 5e-3;
266 static double mean = 5e-3;
267 static double m2 = 0;
268 static int64_t count = 1;
269 auto seconds = ms / 1000.;
270
271 while (seconds > estimate) {
272 auto start = std::chrono::high_resolution_clock::now();
273 std::this_thread::sleep_for(std::chrono::milliseconds(1));
274 auto end = std::chrono::high_resolution_clock::now();
275
276 double observed = (end - start).count() / 1e9;
277 seconds -= observed;
278
279 ++count;
280 double delta = observed - mean;
281 mean += delta / count;
282 m2 += delta * (observed - mean);
283 double stddev = sqrt(m2 / (count - 1));
284 estimate = mean + stddev;
285 }
286
287 // spin lock
288 auto start = std::chrono::high_resolution_clock::now();
289 while ((std::chrono::high_resolution_clock::now() - start).count() / 1e9 < seconds)
290 ;
291}

309template<typename T, typename U, typename std::enable_if<std::is_arithmetic<T>::value> * = nullptr>
310auto narrow(U u) -> T {
311 constexpr bool is_different_signedness = (std::is_signed<T>::value != std::is_signed<U>::value);
312 const T t = narrow<T>(u);
313
314 if (static_cast<U>(t) != u || (is_different_signedness && ((t < T{}) != (u < U{})))) {
315 DSP_THROW_EXCEPTION(std::out_of_range("dsp::narrow"));
316 }
317
318 return t;
319}
