221 template<typename T>
222 auto get_all_objects(uint16_t tid, size_t max_sz) -> vector<T> {
223
224 auto count = p_conn->get_lv_of_class(tid);
225 if (count <= 0) { return vector<T>{}; }
226
227 auto res = dsp::make_empty_objs<T>(count);
228
229 auto data_sz = sizeof(T) * count;
230 if (data_sz <= max_sz) {
231 p_conn->get_all_objects(tid, count, res.data(), data_sz);
232 return res;
233 }
234
235 size_t once_count = std::floor(max_sz / sizeof(T));
236 size_t x = count / once_count;
237 size_t y = count % once_count;
238
239 for (size_t i = 0; i < x; i++) {
240 p_conn->get_objects_by_sub(tid,
241 once_count * i,
242 once_count,
243 res.data() + once_count * i,
244 sizeof(T) * once_count);
245 }
246 p_conn->get_objects_by_sub(tid, once_count * x, y, res.data() + once_count * x, sizeof(T) * y);
247
248 return res;
249 }

326 template<typename T>
327 auto get_related_objects(const OID &oid, uint16_t fid, size_t max_sz = DEFAULT_MAX_MSG_SIZE) -> vector<T> {
328
329 auto info = get_attrs_info_by_id(oid.db_class_id, fid);
330 size_t count{0};
331 switch (info.type_id) {
332 case DT_S_REF_SIM:
333 case DT_S_REF_REL:
334 case DT_S_REF_INT:
335 case DT_S_REF_MATE: {
336 count = 1;
337 break;
338 }
339 case DT_S_DARRAY_COMP:
340 case DT_S_ARRAY_COMP: {
341 auto comp = get_attr_by_oid<s_array_composition>(oid, fid);
342 count = comp.coll_num;
343 break;
344 }
345 case DT_S_MARRAY_COMP: {
346 auto comp = get_attr_by_oid<s_Marray_composition>(oid, fid);
347 count = comp.coll_num;
348 break;
349 }
350 default: {
351 break;
352 }
353 }
354
355 if (count == 0) { return vector<T>{}; }
356
357 auto res = dsp::make_empty_objs<T>(count);
358 size_t data_sz = sizeof(T) * count;
359 if (data_sz <= max_sz) {
360 p_conn->get_related_objects(oid, fid, res.data(), data_sz, 0, count);
361 return res;
362 }
363
364 size_t once_count = std::floor(max_sz / sizeof(T));
365 size_t x = count / once_count;
366 size_t y = count % once_count;
367 for (int i = 0; i < x; ++i) {
368 p_conn->get_related_objects(oid,
369 fid,
370 res.data() + i * once_count,
371 sizeof(T) * once_count,
372 i * once_count,
373 once_count);
374 }
375 p_conn->get_related_objects(oid, fid, res.data() + x * once_count, sizeof(T) * y, x * once_count, y);
376
377 return res;
378 }

441 auto get_integer_attr_by_oid(const OID &oid, uint16_t fid, bool *ok = nullptr) -> int32_t {
442 if (ok) { *ok = true; }
443
444 auto field_info = get_attrs_info_by_id(fid);
445 if (dsp::is_in(field_info.field_id, std::vector<uint16_t>{DT_BOOL, DT_CHAR, DT_BYTE, DT_BIT})) {
446 return get_attr_by_oid<bool>(oid, fid);
447 }
448
449 if (dsp::is_in(field_info.field_id, std::vector<uint16_t>{DT_ENUM, DT_LONG, DT_ULONG})) {
450 return get_attr_by_oid<int32_t>(oid, fid);
451 }
452
453 if (dsp::is_in(field_info.field_id, std::vector<uint16_t>{DT_SHORT, DT_USHORT})) {
454 return get_attr_by_oid<int16_t>(oid, fid);
455 }
456
457 if (ok) { *ok = false; }
458
459 return 0;
460 }
