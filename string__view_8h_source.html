448 auto find(basic_string_view s, size_type pos = 0) const noexcept -> size_type {
449 if (pos > size()) { return npos; }
450
451 if (s.empty()) { return pos; }
452
453 if (s.size() > size() - pos) { return npos; }
454
455 const charT *cur = ptr_ + pos;
456 const charT *last = cend() - s.size() + 1;
457 for (; cur != last; ++cur) {
458 cur = traits::find(cur, last - cur, s[0]);
459 if (!cur) { return npos; }
460 if (traits::compare(cur, s.cbegin(), s.size()) == 0) { return cur - ptr_; }
461 }
462 return npos;
463 }

512 auto _split(basic_string_view k, bool keep_empty_parts = false) -> std::vector<basic_string_view> {
513 std::vector<basic_string_view> res{};
514
515 size_type begin_pos = 0;
516 size_type end_pos = 0;
517 while (true) {
518 end_pos = find(k, begin_pos);
519 if (end_pos == npos) {
520 res.emplace_back(substr(begin_pos));
521 break;
522 }
523
524 DEFER { begin_pos = end_pos + k.size(); };
525
526 if (begin_pos == end_pos && !keep_empty_parts) { continue; }
527 res.emplace_back(substr(begin_pos, end_pos).data(), (end_pos - begin_pos));
528 }
529
530 return res;
531 }

556 static auto join(const std::vector<basic_string_view> &string_views,
557 const std::basic_string<charT, traits> &sep = "") -> std::basic_string<charT, traits> {
558 size_t sz{0};
559 for (auto &sv : string_views) { sz += sv.size(); }
560 sz += (string_views.size() - 1) * sep.size();
561
562 std::basic_string<charT, traits> str(sz, '0');
563
564 charT *data = &str[0];
565 for (size_t i = 0; i < string_views.size(); ++i) {
566 memcpy(data, string_views[i].data(), string_views[i].size());
567 data += string_views[i].size();
568
569 if (i != string_views.size() - 1 && !sep.empty()) {
570 memcpy(data, sep.data(), sep.length());
571 data += sep.length();
572 }
573 }
574
575 return str;
576 }

589 static auto join(const std::vector<std::basic_string<charT, traits>> &strings,
590 const std::basic_string<charT, traits> &sep = "") -> std::basic_string<charT, traits> {
591 std::vector<basic_string_view> views(strings.size());

593 strings.end(),
594 views.begin(),
595 [](const std::basic_string<charT, traits> &str) -> basic_string_view {
596 return basic_string_view(str);
597 });
598
599 return join(views, sep);

804 auto rfind(basic_string_view s, size_type pos = npos) const noexcept -> size_type {
805 if (len_ < s.len_) { return npos; }
806 if (pos > len_ - s.len_) pos = len_ - s.len_;
807 if (s.len_ == 0u) {
808 // an empty string is always found
809 return pos;
810 }
811 for (const charT *cur = ptr_ + pos;; --cur) {
812 if (traits::compare(cur, s.ptr_, s.len_) == 0) { return cur - ptr_; }
813 if (cur == ptr_) { return npos; }
814 };
815 }
